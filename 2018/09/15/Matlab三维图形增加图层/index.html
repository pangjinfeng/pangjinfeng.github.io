Matlab三维图形增加图层

发表于 2018-09-15

%% 圆柱
R=0.1;
h=1.7;
m=100;
[x,y,z]=cylinder(R,m);%创建以(0,0)为圆心，高度为[0,1]，半径为R的圆柱
% z = z*h %高度为h
meshz(x,y,z)

%% 增加两条直线 （增加图层方法一致）
hold on
x1 = [-0.11,-0.11];
x2 = [-0.12,-0.17];
y = [0.1,0.1];
z = [0,1.7];
plot3(x1,y,z);
plot3(x2,y,z);
hold off
%%
